  1. Is there really a home advantage? Use a box plot to show the number of goals scored by home and away teams.
raw_goals <- dbGetQuery(con, "select home_team_goal, away_team_goal from match")

plot_data <- raw_goals %>%
  pivot_longer(cols = everything(), 
               names_to = "Location", 
               values_to = "Goals") %>%
  mutate(Location = if_else(Location == "home_team_goal", "Home", "Away"))

ggplot(plot_data, aes(x = Location, y = Goals, fill = Location)) +
  geom_boxplot(alpha = 0.7) +
  labs(title = "Graph",
       y = "Number of Goals") +
  theme_minimal()

goals_and_dates <- dbGetQuery(con, "select home_team_goal, away_team_goal, date 
from match")

  mutate(date = as_date(date), month_number = month(date),
         total_goals = home_team_goal + away_team_goal) %>%
  group_by(month_number) %>%
  summarise(avg_goals=mean(total_goals, na.rm = TRUE))
